import mongoose from 'mongoose';

/**
 * Retro Model to store the different fields of each retro.
 */
const retroSchema = new mongoose.Schema({
  title: {
    type: String,
    required: [true, 'Please add a title'],
  },
  date: {
    type: Date,
    required: [true, 'Please select a date'],
  },
  overview: {
    type: String,
    required: [true, 'Please provide an overview'],
  },
  technicalContributions: {
    type: String,
    required: [true, 'Please add technical contributions'],
  },
  teamContributions: {
    type: String,
    required: [true, 'Please add team contributions'],
  },
  widerContributions: {
    type: String,
    required: [true, 'Please add wider contributions'],
  },
  improvementAndReflections: {
    type: String,
    required: [true, 'Please add improvements and reflections'],
  },
  tags: {
    type: [String],
    required: [true, 'Please add tags'],
  },
  overallFeeling: {
    type: Number,
    required: [true, 'Please provide your overall feeling'],
  },
});

module.exports = mongoose.model('Retros', retroSchema);
