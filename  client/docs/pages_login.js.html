import LoginForm from '../components/LoginForm';
import SignUpForm from '../components/SignUpForm';
import Logo from '../components/Logo';
import styled from 'styled-components';
import { useState } from 'react';

/**
 * Page Component for the Login/Sign up Page
 * @returns Logo, Sign Up and Login forms, and a button to switche between the forms.
 */
function Login() {

  /**
   * Declare default component state variables
   */
  const [forms, setForms] = useState ({
    signup:true,
    login: false
  });

  /**
   * Handler for switching states between signup and login
   */
  const handleForm = () => {
    setForms((prevState) => ({
      signup: !prevState.signup,
      login: !prevState.login
    }))

  }
  return (
    <>
      <Logo />
      <StyledContainerDiv>
        {forms.login && <LoginForm />}
        {forms.signup && <SignUpForm />}
        <br />
        <button onClick={() => handleForm()}>
          {forms.signup ? 'Already a member? Login' : 'Not a member? Sign up'}
        </button>
      </StyledContainerDiv>
    </>
  );
}

export default Login;

  flex-direction: column;
  padding-top: 10%;
  height:100%;
  display: flex;
  align-items:center;
  justfiy-content: center;
